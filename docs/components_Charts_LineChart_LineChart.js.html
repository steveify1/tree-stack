import React from 'react';
import { Line } from 'react-chartjs-2';
import { string, number, shape as ObjectShape, arrayOf } from 'prop-types';

/**
 * Dataset - this specifies the `data` representation itself for the `LineChart` component
 * @see <a href="/docs/LineChart.html">LineChart parent component</a>
 *
 * @typedef {object} Dataset
 * @property {string} label - this is for the title of the the data to be represented
 * @property {Array<number>} data - each item in this `Array` is a `data` representation for each item in the `labels` array in the `LineChartData typedef`
 * @see <a href="/docs/LineChart.html">LineChart typedef</a>
 *
 * @property {string} [borderWidth] - (`optional`) for the width of the `line-path` in the chart. eg: "1.5"... (note: don't provide with "px". eg: "1.5px"!!!);
 * @property {Array<string>} [borderColor] - (`optional`) an `Array` for the border-color of the `line-path` in the chart. eg: ['red']...
 * @property {Array<string>} [backgroundColor] - (`optional`) an `Array` for the background-color of the `fill` in the chart. eg: ['orange']...
 * @property {string} [pointBorderColor] - (`optional`) for the border-color of the `point` in the chart. eg: 'blue'...
 * @property {string} [pointBackgroundColor] - (`optional`) for the background-color of the `point` in the chart. eg: 'cyan'...
 */

/**
 * LineChartData - this is to be passed as props to the LineChart component
 * @see <a href="/docs/LineChart.html">LineChart component</a>
 *
 * @typedef {object} LineChartData
 * @property {Array<string>} labels - (`required`) for the `labels` of the `xAxes` in the `LineChart`
 * @property {Array<Dataset>} datasets - each object correspond to 1 `line-path` and `fill` in the chart
 *
 */

/**
 * LineChart component - for visualizing data in a Line-chart format: (charts Component)
 *
 * @component
 * @param props
 * @param {LineChartData} props.data
 * @param {number} props.width - for customizing the `width` of the `chart`. `note` the `higher` the `number` passed the `smaller` the `height` will be.
 * @param {number} props.height - for customizing the `height` of the `chart`.
 *
 * @example
 * USAGE
 *
 * const lineChartProps = {
 * 	width: 100,
 * 	height: 50,
 *
 * 	data: {
 * 		labels: ['jan', 'feb', 'mar'],
 *
 * 		datasets: [
 * 			{
 * 				label: 'sales for 2020 (in millions)',
 * 				data: [3, 2, 2],
 *					borderColor: ['rgb(57, 21, 154)'],
 * 				backgroundColor: ['rgba(57, 21, 154, .1)'],
 * 				pointBorderColor: 'rgb(225, 225, 225)',
 * 				pointBackgroundColor: 'rgb(57, 21, 154)',
 * 				borderWidth: '1.5',
 * 			}
 * 		],
 * 	},
 * }
 *
 * return (
 * 	<LineChart {...lineChartProps} />
 * )
 */
function LineChart(props) {
	const { data, width, height, ...rest } = props;

	const lineProps = {
		data,

		options: {
			// 	title: {
			// 		text: 'line chart',
			// 		display: true,
			// 	},
			scales: {
				yAxes: [
					{
						ticks: {
							min: 0,
							max: 6,
							stepSize: 1,
						},
					},
				],
			},
		},
	};

	// small boxes
	// return <Line width={500} height={500} {...lineProps} />;
	return <Line width={width} height={height} {...lineProps} {...rest} />;
}

LineChart.propTypes = {
	data: ObjectShape({
		labels: arrayOf(string).isRequired,
		datasets: arrayOf(
			ObjectShape({
				label: string.isRequired,
				data: arrayOf(number).isRequired,
				borderColor: arrayOf(string),
				backgroundColor: arrayOf(string),
				pointBorderColor: string,
				pointBackgroundColor: string,
				borderWidth: string,
			})
		).isRequired,
	}).isRequired,

	/**
	 * for customizing the `width` of the `chart`. `note` the `higher` the `number` passed the `smaller` the `height` will be.
	 */
	width: number,
	/**
	 * for customizing the `height` of the `chart`.
	 */
	height: number,
};

export default LineChart;
