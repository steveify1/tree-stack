import React from 'react';
import './MainContent.scss';
import { Switch, Route, Redirect } from 'react-router-dom';
import {
	Dashboard,
	NotFound,
	Articles,
	Collections,
	Sequels,
	Projects,
	Reviews,
	Schedules,
	Support,
	Tutorials,
} from '../../pages';

/**
 * MainContent Component - a direct child of "Main" component, where the main content:(composed routed pages component/s) of the application is displayed: (layout Component)
 * @see <a href="/docs/Main.html">Main (parent component)</a>
 *
 * @see <a href="/docs/Dashboard.html">Dashboard (child component)</a>
 * @see <a href="/docs/Articles.html">Articles (child component)</a>
 * @see <a href="/docs/Collections.html">Collections (child component)</a>
 * @see <a href="/docs/Sequels.html">Sequels (child component)</a>
 * @see <a href="/docs/Projects.html">Projects (child component)</a>
 * @see <a href="/docs/Reviews.html">Reviews (child component)</a>
 * @see <a href="/docs/Schedules.html">Schedules (child component)</a>
 * @see <a href="/docs/Support.html">Support (child component)</a>
 * @see <a href="/docs/Tutorials.html">Tutorials (child component)</a>
 *
 * @component
 * @example
 * return (
 *  <MainContent>
 *    <Section>the dashboard<Section>
 *  </MainContent>
 * )
 */
function MainContent() {
	return (
		<main className='MainContent' data-test='main-content'>
			<Switch>
				<Route
					path='/dashboard'
					render={(props) => <Dashboard {...props} />}
				/>
				<Route
					path='/articles'
					render={(props) => <Articles {...props} />}
				/>
				<Route
					path='/collections'
					render={(props) => <Collections {...props} />}
				/>
				{/* <Route
					path='/collections/:id'
					render={(props) => <Collections {...props} />}
				/> */}
				<Route path='/Sequels' render={(props) => <Sequels {...props} />} />
				<Route
					path='/projects'
					render={(props) => <Projects {...props} />}
				/>
				<Route path='/Reviews' render={(props) => <Reviews {...props} />} />
				<Route
					path='/schedules'
					render={(props) => <Schedules {...props} />}
				/>
				<Route path='/support' render={(props) => <Support {...props} />} />
				<Route
					path='/tutorials'
					render={(props) => <Tutorials {...props} />}
				/>
				{/**
				 * unhandled routes go to `/not-found` route
				 */}
				<Route path='*' render={(props) => <NotFound {...props} />} />
			</Switch>
		</main>
	);
}

export default MainContent;
