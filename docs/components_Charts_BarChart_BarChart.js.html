import React from 'react';
import { Bar } from 'react-chartjs-2';

function BarChart() {
	const barProps = {
		data: {
			labels: ['jan', 'feb', 'mar', 'apr', 'may'],
			/**
			 * specifies the `data` itself
			 * @type {Array<object>} - each object correspond to 1 `line` in the chart
			 */
			datasets: [
				{
					label: 'sales for 2020 (in millions)',
					data: [3, 2, 2, 1, 5],
					borderWidth: '1.5',
					backgroundColor: [
						'rgba(57, 21, 154, .5)',
						'rgba(57, 21, 154, .5)',
						'rgba(57, 21, 154, .5)',
						'rgba(57, 21, 154, .5)',
						'rgba(57, 21, 154, .5)',
					],
				},
				{
					label: 'sales for 2019 (in millions)',
					data: [1, 2, 5, 2, 3],
					borderWidth: '1.5',
					backgroundColor: [
						'rgba(51, 204, 153, .5)',
						'rgba(51, 204, 153, .5)',
						'rgba(51, 204, 153, .5)',
						'rgba(51, 204, 153, .5)',
						'rgba(51, 204, 153, .5)',
					],
				},
			],
		},

		options: {
			// title: {
			// 	text: 'line chart',
			// 	display: true,
			// },
			scales: {
				yAxes: [
					{
						ticks: {
							min: 0,
							max: 6,
							stepSize: 1,
						},
					},
				],
			},
		},
	};

	return <Bar {...barProps} />;
}

export default BarChart;
